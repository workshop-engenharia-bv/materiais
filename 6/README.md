CREATE STREAM SIMUABAND
WITH (KAFKA_TOPIC = 'negocio-rodas-simulacao-abandonada-veiculo-leve',
      VALUE_FORMAT='AVRO',
      VALUE_SCHEMA_ID=100005) AS

FROM NEGOCIORODASSIMULACAOVEICULOLEVE
